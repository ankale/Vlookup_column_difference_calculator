    <script>

        // Your JavaScript functions

        function calculateDifference() {

            var firstInput = document.getElementById("firstInput").value.toUpperCase();

            var secondInput = document.getElementById("secondInput").value.toUpperCase();


            if (!isValidInput(firstInput) || !isValidInput(secondInput)) {

                alert("Please enter valid alphabetic characters.");

                return;

            }


            var numericValue1 = getNumericValue(firstInput);

            var numericValue2 = getNumericValue(secondInput);


            var difference = Math.abs(numericValue1 - numericValue2 - 1);


            document.getElementById("result").innerHTML = "The difference between " + firstInput + " and " + secondInput + " is " + difference;

        }


        function isValidInput(input) {

            return /^[A-Za-z]+$/.test(input);

        }


        function getNumericValue(input) {

            var numericValue = 0;


            for (var i = 0; i < input.length; i++) {

                numericValue *= 26;

                numericValue += input.charCodeAt(i) - 65 + 1; // Convert character to numeric value (A=1, B=2, ..., Z=26)

            }


            return numericValue;

        }
